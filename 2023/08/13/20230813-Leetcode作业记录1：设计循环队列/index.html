class MyCircularQueue {

    private int head;
    private int tail;
    private int[] data;
    private int len;
    private int count;


    public MyCircularQueue(int k) {
        head = 0;
        tail = 0;
        len = k;
        count = 0;
        data = new int[k];   

    }
    
    public boolean enQueue(int value) {
        if(isFull())    return false;
        else{
                data[tail] = value;
                tail=(tail+1)%len;
                count++;
                return true;
        }
        
    }
    
    public boolean deQueue() {
        if(isEmpty()) return false;
        else{
            head=(head+1)%len;
            count--;
            return true;
        }
    }
    
    public int Front() {
        if(isEmpty()) return -1;
        else{           
            return data[head];
        }
    }
    
    public int Rear() {
        if(isEmpty()) return -1;
        else return data[(tail+len-1)%len];
        
    }
    
    public boolean isEmpty() {

        if( count==0)
            return true;
        else 
            return false; 
              

    }
    
    public boolean isFull() {
        if(count == len)
            return true;
        else 
            return false;    

    }
}
