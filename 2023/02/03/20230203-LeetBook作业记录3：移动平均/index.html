class MovingAverage {

    /** Initialize your data structure here. */
    int size;
    int[] data;
    int head;
    int tail;
    int count;

    public MovingAverage(int size) {
        data = new int[size];
        head = 0;
        tail = 0;
        this.size = size;
        count = 0;

    }
    
    public double next(int val) {
        double avg = 0;

        if(count == 0){
            data[head] = val;
            tail = (tail+1)%size;
            count++;
        }else  if(count == size){   
            head = (head+1)%size;
            data[tail]=val;
            tail = (tail+1)%size;
        }else{
            data[tail]=val;
            tail = (tail+1)%size;
            count++;
        }
        
        int index = head;
        for(int i = 0; i < count;i++){
            avg = avg + data[index];  
            index = (index+1)%size;        
        }

        return avg/count;

    }
}

class MovingAverage {

    /** Initialize your data structure here. */
    int size;
    int[] data;
    int head;
    int tail;
    int count;
    double sum;

    public MovingAverage(int size) {
        data = new int[size];
        head = 0;
        tail = 0;
        this.size = size;
        count = 0;
        sum = 0;

    }
    
    public double next(int val) {
        double avg = 0;

        if(count == 0){
            data[head] = val;
            sum = sum + val;
            tail = (tail+1)%size;
            count++;
        }else if(count == size){ 
            sum = sum - data[head];  
            tail = head;
            data[tail] = val;
            head = (head+1)%size;
            sum = sum + val;
        }else{
            data[tail]=val;
            tail = (tail+1)%size;
            count++;
            sum = sum + val;
        }
        
        

        return avg=sum/count;

    }
}

 * MovingAverage obj = new MovingAverage(size);
